Documentación

{
    public function __toString() {
        return "Soy una clase";
    }

class Demo
{
    public function __invoke($param) {
        echo "Me han invocado con este parámetro: ".$param;

$clase(25); // Me han invocado con este parámetro: 25
class Demo
{
    private $propiedad = 5;
	
    public function __debugInfo() {
        return [
            'propiedad' => $this->propiedad * 5
        ];
    }
}

$clase = new Demo();
var_dump($clase);
// object(Demo)#1 (1) { ["propiedad"] => int(25) }

G - Generators

 

Documentación

function recorrerUnMillon() {
    for ($i = 0; $i < 1000000; $i++) {
        yield $i;
    }						
}

foreach (recorrerUnMillon() as $numero) {
    if ($numero == 10) {
        break;
    }

    echo $numero;
}
// 0123456789

J - Funciones variadicas

 

Documentación

function sumar(...$numeros) {
    $suma = 0;

    foreach ($numeros as $numero) {
        $suma += $numero;
    }
	
    return $suma;
}

echo sumar(1, 2); // 3
echo sumar(1, 2, 3, 4); // 10

        return null;

    return strtoupper($parametro);
}

echo transformar("a"); // "A"

L - Operador null coalescing

 

Documentación

// esto
$valor = isset($lista['valor']) ? $lista['valor'] : "desconocido";
echo $valor;

// es igual a esto
$valor = $lista['valor'] ?? "desconocido";
echo $valor;
